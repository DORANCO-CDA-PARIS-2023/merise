Affichez les étudiants qui ont emprunté un livre après une date spécifique.

	FROM student AS st
	JOIN borrow ON st.id = borrow.id_student

Affichez le nom et le prénom de l'auteur de chaque livre.
	SELECT book.title, author.name AS nom_auteur, author.firstname AS prenom_auteur 
	from book
	join author on book.id_author = author.id;


Affichez les livres empruntés par un étudiant spécifique en incluant les détails de l'emprunt.
	SELECT book.titre, borrow.Date_borrow, borrow.Date_back_schedulde
	FROM book
	JOIN borrow ON book.id_livre = borrow.id_book
	JOIN borrow ON borrow.id_student = student.id
	WHERE student.nom = 'NomEtudiant' AND student.name = 'Leclerc';

	select count(*) from book;


Affichez le nombre total de livres écrits par chaque auteur.
	SELECT author.name AS nom_auteur, author.firstname AS prenom_auteur, 	COUNT(book.id) AS nombre_de_livres
	FROM author
	JOIN book ON author.id = book.id_author
	GROUP BY author.id, author.name, author.firstname;

Affichez le nombre d'emprunts effectués par chaque étudiant.
	SELECT student.name, student.firstname, COUNT(borrow.id) AS nombre_emprunts
	FROM student
	JOIN borrow ON student.id = borrow.id_student
	GROUP BY student.id, student.name, student.firstname;
